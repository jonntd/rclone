```go
func (ucv *UnifiedCacheViewer) extractDirIDFromKey(key string) string {
    // dirlist_2534907254389549917_ -> 2534907254389549917
    parts := strings.Split(key, "_")
    if len(parts) >= 2 && parts[1] != "" {
        return parts[1]
    }
    return ""
}
```
